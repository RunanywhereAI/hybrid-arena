const yaml = require('yaml')

function parseFrontMatter(noteContent) {
  if (!noteContent || typeof noteContent !== 'string') {
    return { attributes: {}, body: noteContent || '' }
  }

  const lines = noteContent.split('\n')
  let frontMatterLines = []
  let i = 0
  let inFrontMatter = false

  // Check if the first line is the start of front-matter
  if (lines[0] === '---') {
    inFrontMatter = true
    i = 1
  }

  // Collect front-matter lines
  while (i < lines.length) {
    if (lines[i] === '---') {
      inFrontMatter = false
      i++
      break
    }
    if (inFrontMatter) {
      frontMatterLines.push(lines[i])
    }
    i++
  }

  // Parse front-matter if found
  let attributes = {}
  if (frontMatterLines.length > 0) {
    try {
      attributes = yaml.parse(frontMatterLines.join('\n'))
    } catch (e) {
      // If YAML parsing fails, return empty attributes
      attributes = {}
    }
  }

  // Get the remaining body text
  const body = lines.slice(i).join('\n').trim()

  return { attributes, body }
}

module.exports = { parseFrontMatter }

// src/storage.js
const sqlite3 = require('sqlite3').verbose()
const { open } = require('sqlite')
const path = require('path')

let db

async function initDatabase() {
  if (db) return db
  
  const dbPath = path.join(process.cwd(), 'notes.db')
  db = await open({
    filename: dbPath,
    driver: sqlite3.Database
  })
  
  await db.exec(`
    CREATE TABLE IF NOT EXISTS notes (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      title TEXT,
      body TEXT,
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
      note_date DATE,
      metadata TEXT
    )
  `)
  
  return db
}

async function insertNote(note) {
  const { title, body, note_date, metadata } = note
  const result = await db.run(
    `INSERT INTO notes (title, body, note_date, metadata) VALUES (?, ?, ?, ?)`,
    [title, body, note_date, JSON.stringify(metadata)]
  )
  return result.lastID
}

async function getNotesByDate(date) {
  const notes = await db.all(
    `SELECT * FROM notes WHERE note_date = ? ORDER BY created_at DESC`,
    [date]
  )
  
  return notes.map(note => ({
    ...note,
    metadata: note.metadata ? JSON.parse(note.metadata) : {}
  }))
}

async function getAllNotes() {
  const notes = await db.all(`SELECT * FROM notes ORDER BY created_at DESC`)
  
  return notes.map(note => ({
    ...note,
    metadata: note.metadata ? JSON.parse(note.metadata) : {}
  }))
}

module.exports = {
