#!/usr/bin/env node
'use strict';

const fs = require('node:fs');

function countText(text) {
  const wordCount = text.trim() === '' ? 0 : text.trim().split(/\s+/).length;
  const characterCount = Array.from(text).length;

  const lineTerminators = text.match(/\r\n|\r|\n/g);
  const terminatorCount = lineTerminators ? lineTerminators.length : 0;
  const endsWithLineTerminator = /(?:\r\n|\r|\n)$/.test(text);

  const lineCount =
    text.length === 0 ? 0 : terminatorCount + (endsWithLineTerminator ? 0 : 1);

  const longestLineLength = text
    .split(/\r\n|\r|\n/)
    .reduce((max, line) => Math.max(max, Array.from(line).length), 0);

  return {
    wordCount,
    lineCount,
    characterCount,
    longestLineLength,
  };
}

function main() {
  const filePath = process.argv[2];

  if (!filePath) {
    console.error('Usage: node wordcount.js <path>');
    process.exit(1);
  }

  let text;

  try {
    text = fs.readFileSync(filePath, 'utf8');
  } catch (err) {
    console.error(`Error: cannot read file "${filePath}": ${err.message}`);
    process.exit(1);
  }

  const counts = countText(text);

  console.log(
    [
      counts.wordCount,
      counts.lineCount,
      counts.characterCount,
      counts.longestLineLength,
    ].join('\t')
  );
}

if (require.main === module) {
  main();
}

module.exports = {
  countText,
};

'use strict';

const test = require('node:test');
const assert = require('node:assert/strict');
const fs = require('node:fs');
const os = require('node:os');
const path = require('node:path');
const { spawnSync } = require('node:child_process');

const wordcountPath = path.join(__dirname, 'wordcount.js');

function makeTempDir() {
  return fs.mkdtempSync(path.join(os.tmpdir(), 'wordcount-test-'));
}

function runWordcount(filePath) {
  return spawnSync(process.execPath, [wordcountPath, filePath], {
    encoding: 'utf8',
  });
}

test('normal multi-line file case', () => {
  const dir = makeTempDir();
  const filePath = path.join(dir, 'sample.txt');

  fs.writeFileSync(
    filePath,
    'hello world\nthis is a test\nlongest line here\n',
    'utf8'
  );

  const result = runWordcount(filePath);

  assert.equal(result.status, 0);
  assert.equal(result.stderr, '');
  assert.equal(result.stdout, '9\t3\t45\t17\n');
});

test('empty-file case', () => {
  const dir = makeTempDir();
  const filePath = path.join(dir, 'empty.txt');

  fs.writeFileSync(filePath, '', 'utf8');

  const result = runWordcount(filePath);

  assert.equal(result.status, 0);
  assert.equal(result.stderr, '');
  assert.equal(result.stdout, '0\t0\t0\t0\n');
});

test('missing-file case', () => {
  const dir = makeTempDir();
  const filePath = path.join(dir, 'does-not-exist.txt');

  const result = runWordcount(filePath);

  assert.equal(result.status, 1);
  assert.equal(result.stdout, '');
  assert.match(result.stderr, /Error: cannot read file/);
  assert.match(result.stderr, /does-not-exist\.txt/);
});

# wordcount

A small Node.js CLI tool that prints word count, line count, character count, and longest line length for a text file.

Uses only the Node.js standard library.

## Install

Drop `wordcount.js` into a folder.

Optional: make it executable on macOS/Linux:

```sh
chmod +x wordcount.js
```

## Usage

```sh
node wordcount.js <path>
```

Example:

```sh
node wordcount.js sample.txt
```

Sample output:

```txt
9	3	45	17
```

The output columns are:

```txt
word count	line count	character count	longest line length
```

## Run tests

```sh
node --test wordcount.test.js
```
